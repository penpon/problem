💡 データ管理の基本

配列を使ってデータを管理する基本的な方法：

// お気に入り商品を保存する配列 let favorites = []; // 商品オブジェクトを配列に追加 function addToFavorites(name, price) { let product = { name: name, price: price }; favorites.push(product); // 配列に追加 } // 配列から商品を削除 function removeFromFavorites(name) { favorites = favorites.filter(item => item.name !== name); } // 配列の要素数を確認 let count = favorites.length;
